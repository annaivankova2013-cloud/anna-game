// Створюємо карту const graphics = this.add.graphics(); // Трава graphics.fillStyle(0x7ec850, 1); graphics.fillRect(0, 0, mapWidth, mapHeight); // Дороги graphics.fillStyle(0x555555, 1); graphics.fillRect(0, 600, mapWidth, 80);
